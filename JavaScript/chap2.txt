----------リスト2-17
for(var i = 3;i < 100;i++){
  if (isPrime(i)){
    document.write(i);
    document.write(", ");
  }
}


function isPrime(num){
  for(var j = 2;j < num;j++){
    if (num % j == 0) return false;
  }
  return true;
}
